### 3.1 单元测试（Jest）

```typescript
// examples/stage10-production/02-testing/user.service.spec.ts

import { Test, TestingModule } from '@nestjs/testing';
import { UserService } from './user.service';
import { UserRepository } from './user.repository';
import { NotFoundException } from '@nestjs/common';

describe('UserService', () => {
  let service: UserService;
  let userRepo: jest.Mocked<UserRepository>;

  const mockUser = {
    id: 1,
    username: '张三',
    email: 'zhangsan@example.com',
    age: 25,
    status: 'active' as const,
    createdAt: new Date(),
    updatedAt: new Date(),
  };

  beforeEach(async () => {
    const mockUserRepo = {
      findById: jest.fn(),
      findByEmail: jest.fn(),
      create: jest.fn(),
      update: jest.fn(),
      softDelete: jest.fn(),
    };

    const module: TestingModule = await Test.createTestingModule({
      providers: [
        UserService,
        { provide: UserRepository, useValue: mockUserRepo },
      ],
    }).compile();

    service = module.get<UserService>(UserService);
    userRepo = module.get(UserRepository);
  });

  describe('findById', () => {
    it('应该返回用户', async () => {
      userRepo.findById.mockResolvedValue(mockUser);

      const result = await service.findById(1);

      expect(result).toEqual(mockUser);
      expect(userRepo.findById).toHaveBeenCalledWith(1);
    });

    it('用户不存在时抛出 NotFoundException', async () => {
      userRepo.findById.mockResolvedValue(null);

      await expect(service.findById(999)).rejects.toThrow(NotFoundException);
    });
  });

  describe('create', () => {
    it('应该创建用户并返回', async () => {
      const createDto = { username: '张三', email: 'zhangsan@example.com', password: '123456' };
      userRepo.create.mockResolvedValue(mockUser);

      const result = await service.create(createDto);

      expect(result).toEqual(mockUser);
      expect(userRepo.create).toHaveBeenCalledWith(createDto);
    });
  });

  describe('update', () => {
    it('应该更新用户并返回', async () => {
      const updateDto = { nickname: '新昵称' };
      const updatedUser = { ...mockUser, ...updateDto };
      userRepo.findById.mockResolvedValue(mockUser);
      userRepo.update.mockResolvedValue(updatedUser);

      const result = await service.update(1, updateDto);

      expect(result).toEqual(updatedUser);
    });
  });
});
```

```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { INestApplication, HttpStatus } from '@nestjs/common';
import * as request from 'supertest';
import { AppModule } from '../src/app.module';

describe('UserController (e2e)', () => {
  let app: INestApplication;

    const moduleFixture: TestingModule = await Test.createTestingModule({
      imports: [AppModule],
    }).compile();

    app = moduleFixture.createNestApplication();
    app.setGlobalPrefix('api');
    await app.init();
  });

  afterAll(async () => {
    await app.close();
  });

  describe('/api/users (POST)', () => {
    it('应该创建用户', async () => {
      const createDto = {
        username: '测试用户',
        email: 'test@example.com',
        password: 'Password123',
      };

      const response = await request(app.getHttpServer())
        .post('/api/users')
        .send(createDto)
        .expect(HttpStatus.CREATED);

      expect(response.body).toHaveProperty('id');
      expect(response.body.email).toBe(createDto.email);
    });

    it('邮箱格式错误应返回 400', async () => {
      const invalidDto = {
        username: '测试',
        email: 'invalid-email',
        password: '123',
      };

      await request(app.getHttpServer())
        .post('/api/users')
        .send(invalidDto)
        .expect(HttpStatus.BAD_REQUEST);
    });
  });

  describe('/api/users/:id (GET)', () => {
    it('应该获取用户', async () => {
      await request(app.getHttpServer())
        .get('/api/users/1')
        .expect(HttpStatus.OK)
        .expect((res) => {
          expect(res.body).toHaveProperty('id');
        });
    });

    it('用户不存在应返回 404', async () => {

        .get('/api/users/99999')

    });
  });
```

```yaml
# examples/stage10-production/03-cicd/github-actions.yml

name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]

    branches: [main]

  NODE_VERSION: '22'

  # ========== CI：代码检查 + 测试 ==========
  ci:
    runs-on: ubuntu-latest

      - uses: actions/checkout@v4

      - name: Setup Node.js

          node-version: ${{ env.NODE_VERSION }}

      - name: Install dependencies
        run: npm ci

      - name: Lint

      - name: Unit tests

      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info
          fail_ci_if_error: true

  # ========== E2E 测试 ==========
  e2e:
    runs-on: ubuntu-latest
    timeout-minutes: 30
    needs: ci

    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_ROOT_PASSWORD: test
          MYSQL_DATABASE: testdb
        ports:
          - 3306:3306
        options: >-
          --health-cmd="mysqladmin ping"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=5

      redis:
        image: redis:7-alpine
        ports:
          - 6379:6379

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run E2E tests
        run: npm run test:e2e
        env:
          DATABASE_HOST: localhost
          REDIS_HOST: localhost

  # ========== 构建 Docker 镜像 ==========
  build:
    runs-on: ubuntu-latest
    needs: ci
    if: github.ref == 'refs/heads/main'

    permissions:
      contents: read
      packages: write

    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Login to GHCR
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

  # ========== 部署 ==========
  deploy:
    runs-on: ubuntu-latest
    needs: build
    if: github.ref == 'refs/heads/main'
    environment: production

    steps:
      - name: Deploy to server
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.DEPLOY_HOST }}
          username: ${{ secrets.DEPLOY_USER }}
          key: ${{ secrets.DEPLOY_SSH_KEY }}
          script: |
            docker pull ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:${{ github.sha }}
            docker-compose -f docker-compose.prod.yml up -d
            docker image prune -f
```

```typescript
// 添加内存监控
import { Interval } from '@nestjs/schedule';

@Injectable()
export class MemoryMonitorService {
  private readonly logger = new Logger('MemoryMonitor');

  @Interval(60000) // 每分钟
  logMemoryUsage() {
    const usage = process.memoryUsage();
    const heapUsedMB = Math.round(usage.heapUsed / 1024 / 1024);
    const heapTotalMB = Math.round(usage.heapTotal / 1024 / 1024);
    const rssMB = Math.round(usage.rss / 1024 / 1024);

    this.logger.log(`Memory: RSS=${rssMB}MB, Heap=${heapUsedMB}MB/${heapTotalMB}MB`);

    // 内存超过阈值告警
    if (heapUsedMB > 400) {
      this.logger.warn('Memory usage exceeds 400MB, triggering GC hint');
      // 建议：检查是否有大对象未释放
    }
  }
}
```

## 6️⃣ 高并发设计

### 6.1 秒杀系统架构

```
┌─────────────────────────────────────────────────────────────┐
│                    秒杀系统高并发架构                          │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  用户请求 → CDN（静态资源）→ 负载均衡 → API 网关            │
│                                     ↓                        │
│                               限流（每用户1次/秒）            │
│                                     ↓                        │
│                               验证码预校验                    │
│                                     ↓                        │
│                               Redis 库存扣减（原子操作）       │
│                                     ↓                        │
│                               消息队列削峰                    │
│                                     ↓                        │
│                               异步订单创建                    │
│                                     ↓                        │
│                               支付回调通知                    │
│                                                             │
│  关键技术：                                                 │
│  - 预热：秒杀开始前，将库存加载到 Redis                     │
│  - 限流：网关层 + 业务层双重限流                            │
│  - 原子扣减：Redis DECR 原子操作                            │
│  - 消息队列：RabbitMQ / Kafka 异步处理                      │
│  - 乐观锁：数据库层防止超卖                                  │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 6.2 缓存 + 队列 + 多级分流

```typescript
// 秒杀库存扣减
@Injectable()
export class SeckillService {
  constructor(
    private redis: RedisService,
    private mq: RabbitMQService,
  ) {}

  // 1. 预热：将库存加载到 Redis
  async preloadStock(productId: number, stock: number) {
    await this.redis.set(`seckill:stock:${productId}`, stock);
    await this.redis.set(`seckill:ordered:${productId}`, 0);
  }

  // 2. 抢购：Redis 原子扣减
  async grabSeckill(userId: number, productId: number) {
    const stockKey = `seckill:stock:${productId}`;
    const orderedKey = `seckill:ordered:${productId}`;
    const userOrderKey = `seckill:user:${productId}:${userId}`;

    // 检查用户是否已下单
    const alreadyOrdered = await this.redis.get(userOrderKey);
    if (alreadyOrdered) {
      return { success: false, message: '已参与过秒杀' };
    }

    // 原子扣减库存
    const stock = await this.redis.decr(stockKey);

    if (stock < 0) {
      // 库存不足，恢复库存
      await this.redis.incr(stockKey);
      return { success: false, message: '库存不足' };
    }

    // 标记用户已下单（防止重复）
    await this.redis.setex(userOrderKey, 3600, '1');

    // 发送消息到队列，异步创建订单
    await this.mq.emit('seckill.order', { userId, productId, stock });

    return { success: true, message: '抢购成功' };
  }

  // 3. 异步创建订单（消费者）
  @EventPattern('seckill.order')
  async handleSeckillOrder(data: { userId: number; productId: number; stock: number }) {
    // 创建订单
    const order = await this.orderService.create({
      userId: data.userId,
      productId: data.productId,
      status: 'pending',
    });

    // 更新库存计数
    await this.redis.incr(`seckill:ordered:${data.productId}`);

    // 发送延迟消息，30分钟未支付则取消
    await this.mq.emit('seckill.order.timeout', {
      orderId: order.id,
      productId: data.productId,
    }, { delay: 30 * 60 * 1000 });

    return order;
  }
}
```
