## 1️⃣ JWT 认证原理

### 1.1 JWT 结构

```
┌─────────────────────────────────────────────────────────────┐
│                      JWT Token 结构                          │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  Header.Payload.Signature                                   │
│  ┌─────────┐ ┌─────────────────────────┐ ┌────────────────┐ │
│  │  Header │ │      Payload            │ │   Signature    │ │
│  │  (JSON) │ │  (Claims / 声明)        │ │   (签名)        │ │
│  └────┬────┘ └────────────┬────────────┘ └───────┬────────┘ │
│       │                   │                      │          │
│       ↓                   ↓                      ↓          │
│  {                         {                      HMAC.SHA256(
│    "alg": "HS256",          "sub": "12345",        base64UrlEncode(header) + "." +
│    "typ": "JWT"             "name": "张三",         base64UrlEncode(payload),
│  }                          "role": "admin",       SECRET_KEY
│                              "iat": 1516239022,   )
│                              "exp": 1516242622    )
│                            }                        │
│                                                             │
│  base64UrlEncode(Header).base64UrlEncode(Payload).Signature │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 1.2 Header（头部）

```json
{
  "alg": "HS256",    // 签名算法：HS256（对称）/ RS256（非对称）
  "typ": "JWT"       // Token 类型
}
```

### 1.3 Payload（载荷）

```json
{
  "sub": "1234567890",        // subject：用户 ID
  "name": "张三",              // 自定义声明
  "role": ["admin", "vip"],    // 角色列表
  "permissions": ["user:read", "user:write"],
  "iat": 1516239022,           // issued at：签发时间
  "exp": 1516242622,           // expiration time：过期时间
  "nbf": 1516239022,           // not before：生效时间
  "jti": "unique_token_id"     // jwt id：Token 唯一标识
}
```

### 1.4 HS256 vs RS256

```
┌─────────────────────────────────────────────────────────────┐
│                    签名算法对比                               │
├──────────────────────┬──────────────────────────────────────┤
│       HS256          │              RS256                   │
│    （对称加密）        │           （非对称加密）              │
├──────────────────────┼──────────────────────────────────────┤
│  算法：HMAC + SHA256  │  算法：RSA + SHA256                  │
│  密钥：共享密钥        │  密钥：公钥 + 私钥                    │
│  签名/验证：同一密钥   │  签名：用私钥，验证：用公钥           │
│  速度：快              │  速度：慢（涉及大数运算）              │
│  密钥传输：需要安全渠道 │  密钥传输：公钥可公开，私钥需保护     │
│  适用：内部服务        │  适用：跨服务、第三方验证             │
└──────────────────────┴──────────────────────────────────────┘

示例：
- HS256：微服务内部通信，单一签发方
- RS256：OAuth2、第三方登录、SSO
```

## 2️⃣ NestJS + JWT 实现

### 2.1 JWT 模块配置

```typescript
// examples/stage5-auth-rbac/01-jwt/auth.module.ts

import { Module } from '@nestjs/common';
import { JwtModule } from '@nestjs/jwt';
import { PassportModule } from '@nestjs/passport';
import { AuthController } from './auth.controller';
import { AuthService } from './auth.service';
import { JwtStrategy } from './jwt.strategy';
import { UserModule } from '../user/user.module';

@Module({
  imports: [
    PassportModule.register({ defaultStrategy: 'jwt' }),
    JwtModule.register({
      global: true,                   // 全局注册，其他模块直接注入 JwtService
      secret: process.env.JWT_SECRET, // 签名密钥（生产环境使用复杂随机字符串）
      signOptions: {
        expiresIn: '7d',              // Token 7 天过期
        issuer: 'my-app',             // 签发方
        audience: 'my-app-client',    // 接收方
      },
    }),
    UserModule,
  ],
  controllers: [AuthController],
  providers: [AuthService, JwtStrategy],
  exports: [AuthService, JwtModule],
})
export class AuthModule {}
```

### 2.2 JWT 策略

```typescript
// examples/stage5-auth-rbac/01-jwt/jwt.strategy.ts

import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';
import { AuthService } from './auth.service';

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy, 'jwt') {
  constructor(private readonly authService: AuthService) {
    super({
      // 从请求中提取 Token 的方式
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      // 是否忽略过期（生产环境应 false）
      ignoreExpiration: false,
      // 签名密钥
      secretOrKey: process.env.JWT_SECRET,
      // 验证选项
      verifyOptions: {
        algorithms: ['HS256'],        // 只允许 HS256
        issuer: 'my-app',            // 验证签发方
        audience: 'my-app-client',    // 验证接收方
      },
    });
  }

  // JWT 验证通过后执行
  // 返回的对象会挂载到 request.user 上
  async validate(payload: { sub: number; username: string; roles: string[] }) {
    const user = await this.authService.validateUserById(payload.sub);
    if (!user) {
      throw new UnauthorizedException('用户不存在或已被禁用');
    }
    return {
      id: payload.sub,
      username: payload.username,
      roles: payload.roles,
    };
  }
}
```

### 2.3 JWT 认证守卫

```typescript
// examples/stage5-auth-rbac/01-jwt/jwt-auth.guard.ts

import { Injectable, ExecutionContext, UnauthorizedException } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  canActivate(context: ExecutionContext) {
    return super.canActivate(context);
  }

  handleRequest(err: any, user: any, info: any) {
    if (err || !user) {
      const message = info?.name === 'TokenExpiredError'
        ? 'Token 已过期，请重新登录'
        : info?.name === 'JsonWebTokenError'
        ? 'Token 无效'
        : '未登录或登录已失效';
      throw err || new UnauthorizedException(message);
    }
    return user;
  }
}
```

### 2.4 Auth Service

```typescript
// examples/stage5-auth-rbac/01-jwt/auth.service.ts

import { Injectable, UnauthorizedException, BadRequestException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { UserService } from '../user/user.service';
import * as bcrypt from 'bcrypt';

@Injectable()
export class AuthService {
  constructor(
    private readonly jwtService: JwtService,
    private readonly userService: UserService,
  ) {}

  // ========== 用户注册 ==========
  async register(dto: { username: string; email: string; password: string }) {
    // 检查邮箱是否已存在
    const existing = await this.userService.findByEmail(dto.email);
    if (existing) {
      throw new BadRequestException('该邮箱已被注册');
    }

    // 密码加密
    const passwordHash = await bcrypt.hash(dto.password, 12);

    // 创建用户
    const user = await this.userService.create({
      ...dto,
      passwordHash,
    });

    // 生成 Token
    return this.generateToken(user);
  }

  // ========== 用户登录 ==========
  async login(email: string, password: string) {
    const user = await this.userService.findByEmail(email);
    if (!user) {
      throw new UnauthorizedException('用户名或密码错误');
    }

    // 验证密码
    const isPasswordValid = await bcrypt.compare(password, user.passwordHash);
    if (!isPasswordValid) {
      throw new UnauthorizedException('用户名或密码错误');
    }

    // 检查用户状态
    if (user.status === 'banned') {
      throw new UnauthorizedException('账号已被禁用');
    }

    return this.generateToken(user);
  }

  // ========== Token 刷新 ==========
  async refreshToken(refreshToken: string) {
    try {
      // 验证 Refresh Token
      const payload = this.jwtService.verify(refreshToken, {
        secret: process.env.JWT_REFRESH_SECRET,
      });

      // 检查是否为 refresh token
      if (payload.type !== 'refresh') {
        throw new UnauthorizedException('无效的刷新 Token');
      }

      // 获取用户
      const user = await this.userService.findById(payload.sub);
      if (!user || user.status === 'banned') {
        throw new UnauthorizedException('用户不存在或已被禁用');
      }

      return this.generateToken(user);
    } catch (e) {
      throw new UnauthorizedException('刷新 Token 无效或已过期');
    }
  }

  // ========== 生成 Token（Access + Refresh）==========
  private generateToken(user: { id: number; username: string; email: string; roles: string[] }) {
    const payload = { sub: user.id, username: user.username, roles: user.roles };

    return {
      accessToken: this.jwtService.sign(payload, {
        expiresIn: '15m',  // Access Token 短过期
      }),
      refreshToken: this.jwtService.sign(
        { ...payload, type: 'refresh' },
        { expiresIn: '7d', secret: process.env.JWT_REFRESH_SECRET }
      ),
      expiresIn: 15 * 60, // Access Token 剩余有效期（秒）
    };
  }

  async validateUserById(id: number) {
    return this.userService.findById(id);
  }
}
```

## 3️⃣ RBAC 权限模型

### 3.1 RBAC 模型概念

```
┌─────────────────────────────────────────────────────────────┐
│                    RBAC 权限模型                             │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│                    ┌─────────────┐                          │
│                    │    User     │  ← 用户实体              │
│                    └──────┬──────┘                          │
│                           │                                 │
│                      N:M 关系                                │
│                           │                                 │
│                    ┌──────▼──────┐                          │
│                    │    Role     │  ← 角色（如：管理员、VIP） │
│                    └──────┬──────┘                          │
│                           │                                 │
│                      N:M 关系                                │
│                           │                                 │
│                    ┌──────▼──────┐                          │
│                    │  Permission │  ← 权限（如：user:read）   │
│                    └─────────────┘                          │
│                                                             │
│  ┌───────────────────────────────────────────────────────┐  │
│  │  菜单权限（UI 层面）                                    │  │
│  │  角色 A → 菜单：[Dashboard, User List]                 │  │
│  │  角色 B → 菜单：[Dashboard]                            │  │
│  └───────────────────────────────────────────────────────┘  │
│                                                             │
│  ┌───────────────────────────────────────────────────────┐  │
│  │  API 权限（接口层面）                                   │  │
│  │  角色 A → 权限：[user:read, user:write, user:delete]   │  │
│  │  角色 B → 权限：[user:read]                            │  │
│  └───────────────────────────────────────────────────────┘  │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 3.2 数据库设计

```typescript
// ========== RBAC 实体 ==========

@Entity('users')
@Index(['email'], { unique: true })
export class User {
  @PrimaryGeneratedColumn()
  id: number;

  @Column()
  username: string;

  @Column()
  email: string;

  @Column()
  passwordHash: string;

  @Column({ default: 'active' })
  status: string;

  @Column({ type: 'datetime', nullable: true })
  lastLoginAt: Date;

  // 用户-角色：多对多
  @ManyToMany(() => Role, (role) => role.users)
  @JoinTable({
    name: 'user_roles',
    joinColumn: { name: 'user_id', referencedColumnName: 'id' },
    inverseJoinColumn: { name: 'role_id', referencedColumnName: 'id' },
  })
  roles: Role[];

  // 用户-权限：多对多（直接权限，不通过角色）
  @ManyToMany(() => Permission)
  @JoinTable({
    name: 'user_permissions',
    joinColumn: { name: 'user_id', referencedColumnName: 'id' },
    inverseJoinColumn: { name: 'permission_id', referencedColumnName: 'id' },
  })
  directPermissions: Permission[];
}

@Entity('roles')
export class Role {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ unique: true })
  code: string;  // 如：admin, vip, editor

  @Column()
  name: string;  // 如：管理员，VIP 用户，编辑

  @Column({ default: true })
  status: boolean;

  @Column({ type: 'text', nullable: true })
  description: string;

  // 角色-权限：多对多
  @ManyToMany(() => Permission, (permission) => permission.roles)
  @JoinTable({
    name: 'role_permissions',
    joinColumn: { name: 'role_id', referencedColumnName: 'id' },
    inverseJoinColumn: { name: 'permission_id', referencedColumnName: 'id' },
  })
  permissions: Permission[];

  // 角色-用户：多对多（反向）
  @ManyToMany(() => User, (user) => user.roles)
  users: User[];
}

@Entity('permissions')
export class Permission {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ unique: true })
  code: string;  // 如：user:read, user:write, order:delete, system:config

  @Column()
  name: string;   // 如：查看用户，编辑用户

  @Column({ default: 'api' })
  type: 'api' | 'menu' | 'button';  // 权限类型

  @Column({ nullable: true })
  description: string;

  @Column({ nullable: true })
  menuPath: string;  // 如：/users，对应菜单路径

  @Column({ nullable: true })
  apiPath: string;   // 如：GET /api/users，对应 API 路径

  // 权限-角色：多对多（反向）
  @ManyToMany(() => Role, (role) => role.permissions)
  roles: Role[];
}
```

### 3.4 角色守卫

```typescript
// examples/stage5-auth-rbac/02-rbac/roles.guard.ts

import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { ROLES_KEY } from './roles.decorator';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    // 获取路由上配置的角色要求
    const requiredRoles = this.reflector.getAllAndOverride<string[]>(ROLES_KEY, [
      context.getHandler(),   // 路由级别 @Roles
      context.getClass(),    // 控制器级别 @Roles
    ]);

    // 没有配置角色要求，允许访问
    if (!requiredRoles || requiredRoles.length === 0) {
      return true;
    }

    const { user } = context.switchToHttp().getRequest();

    // 检查用户是否拥有所需角色之一
    const hasRole = requiredRoles.some(role => user.roles?.includes(role));

    if (!hasRole) {
      throw new ForbiddenException('您没有权限执行此操作');
    }

    return true;
  }
}
```

### 3.5 RBAC Service

```typescript
// examples/stage5-auth-rbac/02-rbac/rbac.service.ts

import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { User, Role, Permission } from './entities';

@Injectable()
export class RbacService {
  constructor(
    @InjectRepository(User) private userRepo: Repository<User>,
    @InjectRepository(Role) private roleRepo: Repository<Role>,
    @InjectRepository(Permission) private permRepo: Repository<Permission>,
  ) {}

  // ========== 用户权限查询 ==========

  // 获取用户所有权限（角色权限 + 直接权限）
  async getUserPermissions(userId: number): Promise<string[]> {
    const user = await this.userRepo.findOne({
      where: { id: userId },
      relations: ['roles', 'roles.permissions', 'directPermissions'],
    });

    if (!user) return [];

    const rolePermissions = user.roles
      .flatMap(role => role.permissions)
      .map(p => p.code);

    const directPermissions = user.directPermissions.map(p => p.code);

    // 去重
    return [...new Set([...rolePermissions, ...directPermissions])];
  }

  // 获取用户角色
  async getUserRoles(userId: number): Promise<string[]> {
    const user = await this.userRepo.findOne({
      where: { id: userId },
      relations: ['roles'],
    });
    return user?.roles.map(r => r.code) || [];
  }

  // 检查用户是否拥有某权限
  async hasPermission(userId: number, permissionCode: string): Promise<boolean> {
    const permissions = await this.getUserPermissions(userId);
    return permissions.includes(permissionCode);
  }

  // ========== 角色管理 ==========

  // 创建角色
  async createRole(dto: { code: string; name: string; description?: string; permissionCodes?: string[] }) {
    const role = this.roleRepo.create({ code: dto.code, name: dto.name, description: dto.description });
    await this.roleRepo.save(role);

    if (dto.permissionCodes?.length) {
      const permissions = await this.permRepo.findBy({ code: In(dto.permissionCodes) });
      role.permissions = permissions;
      await this.roleRepo.save(role);
    }

    return role;
  }

  // 给用户分配角色
  async assignRolesToUser(userId: number, roleCodes: string[]) {
    const user = await this.userRepo.findOne({
      where: { id: userId },
      relations: ['roles'],
    });
    if (!user) throw new NotFoundException('用户不存在');

    const roles = await this.roleRepo.findBy({ code: In(roleCodes) });
    user.roles = roles;
    await this.userRepo.save(user);
    return user;
  }
}
```

### 3.6 使用示例

```typescript
@Controller('users')
@UseGuards(JwtAuthGuard, RolesGuard)  // 同时使用 JWT 认证 + 角色检查
export class UserController {
  constructor(private readonly rbacService: RbacService) {}

  @Get()
  @Roles('admin', 'editor')          // admin 或 editor 可访问

  @Post()
  @Roles('admin')                     // 只有 admin 可访问

  @Delete(':id')
  @Roles('admin')                     // 只有 admin 可访问
```

### 4.2 CASL Ability Factory

```typescript
// examples/stage5-auth-rbac/03-casl/casl-ability.factory.ts

import { Injectable } from '@nestjs/common';
import { AbilityBuilder, createMongoAbility, ExtractSubjectType, MongoAbility, InferSubjects } from '@casl/ability';
import { User } from '../user/user.entity';
import { Article } from '../article/article.entity';
import { Comment } from '../comment/comment.entity';

// 定义 Subject 类型
type Subjects = InferSubjects<typeof User | typeof Article | typeof Comment>;

// 定义 Action
export enum Action {
  Manage = 'manage',   // 所有操作
  Read = 'read',
  Create = 'create',
  Update = 'update',
  Delete = 'delete',
}

// Ability 类型
export type AppAbility = MongoAbility<[Action, Subjects]>;

@Injectable()
export class CaslAbilityFactory {
  createForUser(user: { id: number; roles: string[]; isOwner?: boolean }) {
    const { can, cannot, build, extend } = new AbilityBuilder<AppAbility>(
      createMongoAbility,
    );

    // ========== 基于角色的权限定义 ==========

    if (user.roles.includes('admin')) {
      // Admin：所有权限
      can(Action.Manage, 'all');
    }

    if (user.roles.includes('editor')) {
      // Editor：可以创建、编辑自己的文章
      can(Action.Create, Article);
      can(Action.Update, Article);
      can(Action.Delete, Article);
    }

    if (user.roles.includes('vip')) {
      // VIP：可以阅读所有付费内容
      can(Action.Read, Article, { isPaid: true });
      can(Action.Read, Article, { isPaid: false });
    }

    // 普通用户
    can(Action.Read, Article, { isPaid: false }); // 只能阅读免费文章
    can(Action.Read, Comment);
    can(Action.Create, Comment);

    // 只能操作自己的数据
    can(Action.Update, User, ['username', 'avatar', 'bio']); // 可修改自己的部分字段
    can(Action.Update, Comment, ['content']);                 // 可修改自己的评论内容
    cannot(Action.Delete, Article);                            // 普通用户不能删除文章

    return build({
      detectSubjectType: (object) =>
        object.constructor as ExtractSubjectType<Subjects>,
    });
  }
}
```

### 4.4 后端字段级权限

```typescript
// NestJS 后端字段过滤
import { Injectable } from '@nestjs/common';
import { CaslAbilityFactory, Action } from './casl-ability.factory';

@Injectable()
export class ArticleService {
  constructor(
    private readonly abilityFactory: CaslAbilityFactory,
  ) {}

  // 过滤用户只能看到自己有权限访问的字段
  filterFields<T extends object>(user: any, action: Action, entity: T): Partial<T> {
    const ability = this.abilityFactory.createForUser(user);

    if (ability.can(action, entity as any)) {
      // 有权限，返回所有字段
      return entity;
    }

    // 无权限，返回过滤后的字段
    return {};
  }

  // 转换为 JSON 时过滤字段
  toJSON(user: any, article: Article): any {
    const ability = this.abilityFactory.createForUser(user);

    const result: any = { id: article.id };

    if (ability.can(Action.Read, article, 'title')) result.title = article.title;
    if (ability.can(Action.Read, article, 'content')) result.content = article.content;
    if (ability.can(Action.Read, article, 'author')) result.author = article.author;

    // 只有付费用户才能看到价格
    if (ability.can(Action.Read, article, 'price')) result.price = article.price;

    return result;
  }
}
```

### 5.3 OAuth2 Service

```typescript
// examples/stage5-auth-rbac/04-oauth2/oauth2.service.ts

import { Injectable, BadRequestException } from '@nestjs/common';
import { UserService } from '../user/user.service';
import { JwtService } from '@nestjs/jwt';

@Injectable()
export class OAuth2Service {
  constructor(
    private readonly userService: UserService,
    private readonly jwtService: JwtService,
  ) {}

  // ========== 验证/创建 OAuth 用户 ==========
  async validateOAuthUser(params: {
    provider: string;      // 'github' | 'google' | 'wechat'
    providerId: string;    // 第三方平台的用户 ID
    username: string;
    email: string;
    avatar?: string;
  }) {
    // 1. 查找是否已存在该第三方用户
    let user = await this.userService.findByOAuth(params.provider, params.providerId);

    if (user) {
      // 已存在，更新信息并返回
      await this.userService.updateOAuthAvatar(user.id, params.avatar);
    } else {
      // 不存在，创建新用户
      user = await this.userService.createOAuthUser({
        ...params,
        // 生成随机密码（用户无法用密码登录）
        passwordHash: await this.generateRandomPassword(),
      });
    }

    // 2. 生成 JWT
    return this.generateToken(user);
  }

  private async generateRandomPassword(): Promise<string> {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    let password = '';
    for (let i = 0; i < 32; i++) {
      password += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    // 使用 bcrypt 加密
    const bcrypt = require('bcrypt');
    return bcrypt.hash(password, 10);
  }

  private generateToken(user: any) {
    const payload = { sub: user.id, username: user.username, roles: user.roles };
    return {
      accessToken: this.jwtService.sign(payload, { expiresIn: '15m' }),
      refreshToken: this.jwtService.sign({ ...payload, type: 'refresh' }, { expiresIn: '7d' }),
    };
  }
}
```

### 6.1 密码加密

```typescript
// examples/stage5-auth-rbac/05-password/password.service.ts

import { Injectable } from '@nestjs/common';
import * as bcrypt from 'bcrypt';

@Injectable()
export class PasswordService {
  // Bcrypt 配置
  private readonly SALT_ROUNDS = 12;  // 越高越安全，越慢

  // 加密密码
  async hash(password: string): Promise<string> {
    return bcrypt.hash(password, this.SALT_ROUNDS);
  }

  // 验证密码
  async verify(password: string, hash: string): Promise<boolean> {
    return bcrypt.compare(password, hash);
  }

  // 检查密码强度
  validateStrength(password: string): { valid: boolean; errors: string[] } {
    const errors: string[] = [];

    if (password.length < 8) errors.push('密码至少8个字符');
    if (!/[A-Z]/.test(password)) errors.push('密码包含大写字母');
    if (!/[a-z]/.test(password)) errors.push('密码包含小写字母');
    if (!/[0-9]/.test(password)) errors.push('密码包含数字');
    if (!/[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]/.test(password)) {
      errors.push('密码包含特殊字符');
    }

    // 检查常见弱密码
    const weakPasswords = ['password', '12345678', 'qwerty', 'admin', 'letmein'];
    if (weakPasswords.includes(password.toLowerCase())) {
      errors.push('密码太简单，请换一个');
    }

    return { valid: errors.length === 0, errors };
  }

  // 密码泄露检测（Have I Been Pwned API）
  async isPwned(password: string): Promise<boolean> {
    // SHA-1 哈希密码
    const hash = await this.hashPassword(password);
    const prefix = hash.slice(0, 5).toUpperCase();
    const suffix = hash.slice(5).toUpperCase();

    // 查询 HIBP API
    const response = await fetch(`https://api.pwnedpasswords.com/range/${prefix}`);
    const data = await response.text();

    // 检查 suffix 是否在返回结果中
    return data.includes(suffix);
  }

  private async hashPassword(password: string): Promise<string> {
    const crypto = require('crypto');
    return crypto.createHash('sha1').update(password).digest('hex');
  }
}
```

### 6.2 登录安全防护

```typescript
// ========== 登录失败次数限制 ==========

@Injectable()
export class LoginAttemptService {
  private attempts = new Map<string, { count: number; lastAttempt: Date }>();
  private readonly MAX_ATTEMPTS = 5;
  private readonly LOCKOUT_MINUTES = 15;

  async isLockedOut(email: string): Promise<boolean> {
    const attempt = this.attempts.get(email);
    if (!attempt) return false;

    const lockoutEnd = new Date(attempt.lastAttempt.getTime() + this.LOCKOUT_MINUTES * 60 * 1000);
    if (new Date() < lockoutEnd) return true;

    // 锁定期已过，清除记录
    this.attempts.delete(email);
    return false;
  }

  async recordFailedAttempt(email: string): Promise<void> {
    const attempt = this.attempts.get(email);
    if (attempt) {
      attempt.count++;
      attempt.lastAttempt = new Date();
    } else {
      this.attempts.set(email, { count: 1, lastAttempt: new Date() });
    }
  }

  async clearAttempts(email: string): Promise<void> {
    this.attempts.delete(email);
  }

  getRemainingLockoutTime(email: string): number {
    const attempt = this.attempts.get(email);
    if (!attempt) return 0;
    const lockoutEnd = new Date(attempt.lastAttempt.getTime() + this.LOCKOUT_MINUTES * 60 * 1000);
    return Math.max(0, Math.ceil((lockoutEnd.getTime() - Date.now()) / 60000));
  }
}

// ========== 登录 Service 集成 ==========

@Injectable()
export class AuthService {
  constructor(
    private readonly loginAttemptService: LoginAttemptService,
    // ...
  ) {}

  async login(email: string, password: string) {
    // 1. 检查是否被锁定
    if (await this.loginAttemptService.isLockedOut(email)) {
      const remaining = this.loginAttemptService.getRemainingLockoutTime(email);
      throw new UnauthorizedException(`账号已锁定，请在 ${remaining} 分钟后重试`);
    }

    // 2. 验证密码
    const user = await this.userService.findByEmail(email);
    if (!user || !await this.passwordService.verify(password, user.passwordHash)) {
      // 记录失败次数
      await this.loginAttemptService.recordFailedAttempt(email);
      throw new UnauthorizedException('用户名或密码错误');
    }

    // 3. 清除失败记录
    await this.loginAttemptService.clearAttempts(email);

    return this.generateToken(user);
  }
}
```

### 7.1 SSO 原理

```
┌─────────────────────────────────────────────────────────────┐
│                    SSO 单点登录原理                           │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  用户首次登录：                                              │
│  1. 用户访问 App1，输入登录信息                              │
│  2. App1 验证成功后，在 App1 保存 Session                    │
│  3. 同时通知 SSO Server：用户已登录                          │
│                                                             │
│  用户访问 App2：                                             │
│  1. 用户访问 App2，App2 发现未登录                           │
│  2. App2 跳转 SSO Server：询问用户是否已登录？                 │
│  3. SSO Server 检查：用户已登录，返回 Token                   │
│  4. App2 用 Token 完成登录                                   │
│                                                             │
│  实现方式：                                                  │
│  - Cookie 共享（同一主域）                                   │
│  - JWT 跨域共享                                             │
│  - CAS 协议                                                 │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 7.2 基于 JWT 的简单 SSO

```typescript
// ========== SSO Service ==========

@Injectable()
export class SsoService {
  constructor(
    private readonly jwtService: JwtService,
    private readonly redis: RedisService,
  ) {}

  // 颁发全局会话 Token
  async createGlobalSession(userId: number, username: string): Promise<string> {
    const token = this.jwtService.sign(
      { sub: userId, username, type: 'sso' },
      { expiresIn: '24h' }
    );

    // 将 Token 存入 Redis，设置过期时间
    await this.redis.set(`sso:session:${token}`, userId.toString(), 24 * 3600);

    return token;
  }

  // 验证全局会话（各应用共享）
  async validateGlobalSession(token: string): Promise<{ userId: number } | null> {
    try {
      const payload = this.jwtService.verify(token);

      if (payload.type !== 'sso') return null;

      // 检查 Redis 中的会话是否存在
      const userId = await this.redis.get(`sso:session:${token}`);
      if (!userId) return null;

      return { userId: parseInt(userId, 10) };
    } catch {
      return null;
    }
  }

  // 注销全局会话
  async destroyGlobalSession(token: string): Promise<void> {
    await this.redis.del(`sso:session:${token}`);
  }
}
```

```typescript
// ========== Token 黑名单 ==========

@Injectable()
export class TokenBlacklistService {
  constructor(private redis: RedisService) {}

  // 将 Token 加入黑名单
  async add(token: string, expiresAt: Date): Promise<void> {
    const ttl = Math.ceil((expiresAt.getTime() - Date.now()) / 1000);
    if (ttl > 0) {
      await this.redis.set(`blacklist:${token}`, '1', ttl);
    }
  }

  // 检查 Token 是否在黑名单
  async isBlacklisted(token: string): Promise<boolean> {
    const result = await this.redis.get(`blacklist:${token}`);
    return result !== null;
  }
}

// 使用：在 JWT Guard 中检查黑名单
@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private tokenBlacklistService: TokenBlacklistService) {
    super();
  }

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const isParentValid = await super.canActivate(context);
    if (!isParentValid) return false;

    const request = context.switchToHttp().getRequest();
    const token = this.extractTokenFromHeader(request);

    if (token && await this.tokenBlacklistService.isBlacklisted(token)) {
      throw new UnauthorizedException('Token 已失效');
    }

    return true;
  }
}
```
